Set objFSO = CreateObject("Scripting.FileSystemObject")
If objFSO.FolderExists("D:\FSO") Then
    print "Folder name already exits"
    objFSO.DeleteFolder("D:\FSO")
Else 
    Set objFolder = objFSO.CreateFolder("D:\FSO")
    print "Folder created successfully!"
End If
