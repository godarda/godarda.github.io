{{ page.title }}
{{ site.vbsfile }}
{%- highlight vb -%}
Set objFSO = CreateObject("Scripting.FileSystemObject")
Set Folder = objFSO.GetFolder("D:\VBScript")
counter = 0
For Each File In Folder.Files
    counter = counter + 1
Next
print "Total Files " & counter
{%- endhighlight -%}
{{ site.output }}

Total Files 16
