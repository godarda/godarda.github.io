{%- highlight vb -%}
Set objFSO = CreateObject("Scripting.FileSystemObject")
If objFSO.FileExists("D:\VBScript\Trial.txt") Then
    print "File name already exits"
    objFSO.DeleteFile("D:\VBScript\Trial.txt")
Else
    Set objFile = objFSO.CreateTextFile("D:\VBScript\Trial.txt")
    print "File created successfully!"
End If
