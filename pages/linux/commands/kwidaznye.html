mkdir command with options
kodingwindow@kw:~$ ls
a.out  kw.cpp   kw.py    Documents  Music     Public     Videos
kw.c   KW.java  Desktop  Downloads  Pictures  Templates  Weekdays

kodingwindow@kw:~$ pwd
/home/kodingwindow

kodingwindow@kw:~$ mkdir Linux
kodingwindow@kw:~$ cd Linux/

kodingwindow@kw:~/Linux$ pwd
/home/kodingwindow/Linux

kodingwindow@kw:~/Linux$ ls
kodingwindow@kw:~/Linux$ mkdir -p Flavours/{Ubuntu/{Ubuntu_18.04,Ubuntu_20.04},Fedora/{Fedora_31,Fedora_32}}
kodingwindow@kw:~/Linux$ ls
Flavours

kodingwindow@kw:~/Linux$ cd Flavours/Ubuntu/
kodingwindow@kw:~/Linux/Flavours/Ubuntu$ ls
Ubuntu_18.04  Ubuntu_20.04

kodingwindow@kw:~/Linux/Flavours/Ubuntu$ cd ..
kodingwindow@kw:~/Linux/Flavours$ ls

kodingwindow@kw:~/Linux/Flavours$ mkdir -m a=r KaliLinux
