cat and wc commands with options
godarda@gd:~$ cat gd.c
#include<stdio.h>
int main()
{
    fork();
    printf("Hello, World!\n");
    return 0;
}

godarda@gd:~$ head -1 gd.c | wc -c
18

godarda@gd:~$ wc gd.c
 7 10 90 gd.c

godarda@gd:~$ wc -c gd.c
90 gd.c

godarda@gd:~$ wc -l gd.c
7 gd.c

godarda@gd:~$ wc -L gd.c
30 gd.c

godarda@gd:~$ wc -m gd.c
90 gd.c

godarda@gd:~$ wc -mw gd.c
10 90 gd.c

godarda@gd:~$ wc -w gd.c
10 gd.c

godarda@gd:~$ nl gd.c
     1  #include<stdio.h>
     2  int main()
     3  {
     4      fork();
     5      printf("Hello, World!\n");
     6      return 0;
     7  }

cat command with options
godarda@gd:~$ cat gd.c
#include<stdio.h>
int main()
{
    fork();
    printf("Hello, World!\n");
    return 0;
}

godarda@gd:~$ cat -E gd.c
#include<stdio.h>$
int main()$
{$
    fork();$
    printf("Hello, World!\n");$
    return 0;$
}$

godarda@gd:~$ cat -n gd.c
     1  #include<stdio.h>
     2  int main()
     3  {
     4      fork();
     5      printf("Hello, World!\n");
     6      return 0;
     7  }

godarda@gd:~$ cat -b gd.c
     1  #include<stdio.h>
     2  int main()
     3  {
     4      fork();
     5      printf("Hello, World!\n");
     6      return 0;
     7  }
