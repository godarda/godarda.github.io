C program to print uppercase alphabets using the ASCII values
{{ site.cfile }}
{%- highlight c -%}
#include 
int main()
{
    int i;
    for (i = 65; i < 91; i++)
    {
        printf("%c ", i);
    }
    printf("\n");
    return 0;
}
{%- endhighlight -%}
