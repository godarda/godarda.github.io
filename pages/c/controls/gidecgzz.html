C program to demonstrate the use of break statement
{{ site.cfile }}
{%- highlight c -%}
#include 
int main()
{
    for (int i = 0; i < 10; i++)
    {
        if (i == 5)
        {
            break;
        }
        printf("%d\n", i);
    }
    return 0;
}
