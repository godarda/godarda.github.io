Logarithms
octave> log(0)
ans = -Inf

octave> log(1)
ans = 0

octave> log1p(0)
ans = 0

octave> log1p(1)
ans = 0.6931

octave> log(-1)
ans =       0 + 3.1416i

octave> reallog(0)
ans = -Inf

octave> reallog(1)

error: reallog: produced complex result
