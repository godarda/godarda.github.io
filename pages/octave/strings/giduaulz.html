godarda@gd:~$ octave
...
octave> gd = "Koding"
gd = Koding

octave> gd = [gd, "window"]
gd = GoDarda

octave> typeinfo(gd)
ans = string

octave> gd(6)
ans = g

octave> gd(2:11)
ans = odingwindo

octave> length(gd)
ans = 12

octave> fliplr(gd)
ans = wodniwgnidoK

octave> toupper(gd)
ans = KODINGWINDOW

octave> tolower(gd)
ans = godarda

octave> sort(gd)
ans = Kddgiinnooww
