>>> strings=['Thomson','1010011','999.99','A1D3T6Z']

'Thomson'

>>> strings[3]
'A1D3T6Z'

>>> strings[-1]
'A1D3T6Z'

>>> strings[-4]
'Thomson'

>>> len(strings)
4

>>> strings[0]='Neil'
>>> strings[0]
'Neil'

>>> strings[-4]
'Neil'

>>> strings
['Neil', '1010011', '999.99', 'A1D3T6Z']

>>> del strings[3]
>>> strings
['Neil', '1010011', '999.99']

>>> strings.append('Edison')
>>> strings
['Neil', '1010011', '999.99', 'Edison']

>>> strings.reverse()
>>> strings
['Edison', '999.99', '1010011', 'Neil']

>>> strings.sort()

['1010011', '999.99', 'Edison', 'Neil']

>>> strings.insert(2,'Linus')

['1010011', '999.99', 'Linus', 'Edison', 'Neil']

>>> strings.remove('999.99')
