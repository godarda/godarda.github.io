Number conversion using int(), bin(), oct(), and hex() functions
>>> n=255
>>> print("Binary",bin(n))
Binary 0b11111111

>>> print("Octal",oct(n))
Octal 0o377

>>> print("Hexadecimal",hex(n))
Hexadecimal 0xff

>>> n="11111111"
>>> print("Decimal",int(n,2))
Decimal 255

>>> print("Octal",oct(int(n,2)))
Octal 0o377

>>> print("Hexadecimal",hex(int(n,2)))
Hexadecimal 0xff

>>> n=0o377
>>> print("Decimal",int(n))
Decimal 255

>>> print("Hexadecimal",hex(n))
Hexadecimal 0xff

>>> n=0Xff
>>> print("Decimal",int(n))
Decimal 255

>>> print("Octal",oct(n))
Octal 0o377

>>> print("Binary",bin(n))
Binary 0b11111111
