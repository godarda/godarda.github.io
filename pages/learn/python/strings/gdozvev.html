String isalnum() method
# Check if all characters are alphanumeric
>>> s="Hello World!"
>>> s.isalnum()
False

>>> s="Hello World"
>>> s.isalnum()
False

>>> s="13HelloWorld13"
>>> s.isalnum()
True
String isdigit() method
# Check if all characters are digits
>>> s="13HelloWorld13"
>>> s.isdigit()
False

>>> s="1313"
>>> s.isdigit()
True

>>> s="13.13"
>>> s.isdigit()
False

# Check if unicode superscript is a digit
>>> s="\u00B2"
>>> s.isdigit()
True
String isdecimal() method
# Check if unicode superscript is a decimal
>>> s="\u00B2"    # to the power 2
>>> s.isdecimal()
False

# Check if characters are decimals
>>> s="0123456789"
>>> s.isdecimal()
True

# Check if unicode fraction is a decimal
>>> s="\u00BD"    # 1/2
>>> s.isdecimal()
False
String isnumeric() method
# Check if unicode superscript is numeric
>>> s="\u00B2"
>>> s.isnumeric()
True

# Check if digits are numeric
>>> s="0123456789"
>>> s.isnumeric()
True

# Check if string representation of fraction is numeric
>>> s="1/2"
>>> s.isnumeric()
False

# Check if unicode fraction is numeric
>>> s="\u00BD"    # 1/2
>>> s.isnumeric()
True
String isspace() method
# Check if string contains only whitespace
>>> s="Hello World!"
>>> s.isspace()
False

>>> s="       "
>>> s.isspace()
True
String istitle() method
# Check if string is title cased
>>> s="Hello World!"
>>> s.istitle()
True

>>> s="hello world!"
>>> s.istitle()
False
String islower() and isupper() methods
# Check if string is lowercase
>>> s="hello world!"
>>> s.islower()
True

# Check if string is uppercase
>>> s="HELLO, WORLD!"
>>> s.isupper()
True

# Check mixed case
>>> s="Hello, World!"
>>> s.islower()
False

>>> s.isupper()
False
