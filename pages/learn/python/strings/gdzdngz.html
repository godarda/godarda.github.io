# Create a string variable
>>> gd="GoDarda"
>>> gd
'GoDarda'

# Get string length
>>> len(gd)
7

# Access character at index 0
>>> gd[0]
'G'

# Access character at index 6
>>> gd[6]
'a'


# Slice with start, stop, step
>>> gd[0:7:1]
'GoDarda'

>> gd[0:7:2]
'GDra'

>>> gd[0:7:3]
'Gaa'

# Reverse the string
>>> gd[::-1]
'adraDoG'


# Slice with step only
>>> gd[::1]
'GoDarda'

>>> gd[::2]
'GDra'

>>> gd[::3]
'Gaa'


# Slice from index to end
>>> gd[1::]
'oDarda'

>>> gd[2::]
'Darda'

>>> gd[3::]
'arda'
