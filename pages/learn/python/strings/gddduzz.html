# Create string with double quotes
>>> "GoDarda"
'GoDarda'

# Create string with single quotes
>>> 'GoDarda'
'GoDarda'

# Print a string
>>> print("GoDarda")
GoDarda

# Print string with escaped single quotes
>>> print("\'GoDarda\'")
'GoDarda'

# Print string with escaped double quotes
>>> print('\"GoDarda\"')
"GoDarda"

# Print string with newline character
>>> print("Go\nDarda")
Go
Darda

# Assign string to a variable
>>> gd="GoDarda"
>>> gd
'GoDarda'

# Get length of the string
>>> len(gd)
7

# Delete the variable
>>> del gd
>>> gd
Traceback (most recent call last):
  File "<stdin>", line 1, in <module>
NameError: name 'gd' is not defined
Strings Concatenation
>>> s1="Go"
>>> s2="Darda"
# Concatenate string with itself
>>> s1+s1
'GoGo'

# Concatenate two strings
>>> s1+s2
'GoDarda'

# Repeat string n times
>>> s1*5
'GoGoGoGoGo'

# Create a tuple of strings
>>> s1,s2
('Go', 'Darda')
Escape Characters
# String with newline escape sequence
>>> gd="Go\nDarda"
>>> gd
'Go\nDarda'

# Print interpreting escape sequences
>>> gd="Go\nDarda"
>>> print(gd)
Go
Darda

# Raw string (ignores escape sequences)
>>> gd=r"Go\nDarda"
>>> print(gd)
Go\nDarda

# Escape double quotes
>>> print('\"GoDarda\"')
"GoDarda"

# String with tab escape sequence
>>> gd="Go\tDarda"
>>> print(gd)
Go	Darda
