String casefold(), capitalize() and title() methods
godarda@gd:~$ python3
...
# Casefold string (stronger lowercase)
>>> gd="GoDaRdA"
>>> gd.casefold()
'godarda'

>>> gd="GODARDA"
>>> gd.casefold()
'godarda'

>>> gd="godarda"
>>> gd.casefold()
'godarda'

# Capitalize first character
>>> gd="go darda"
>>> gd.capitalize()
'Go darda'

# Convert to title case
>>> gd.title()
'Go Darda'
String upper(), lower() and swapcase() methods
# Convert to uppercase
>>> gd="godarda"
>>> gd.upper()
'GODARDA'

# Convert to lowercase
>>> gd="GODARDA"
>>> gd.lower()
'godarda'

# Swap case
>>> gd="GoDarda"
>>> gd.swapcase()
'gOdARDA'
String rjust() and ljust() methods
# Right justify string
>>> gd="GoDarda"
>>> gd.rjust(20)
'             GoDarda'

# Left justify string
>>> gd.ljust(20)
'GoDarda             '
String strip(), rstrip(), and lstrip() methods
# Remove leading and trailing whitespace
>>> gd="    GoDarda           "
>>> gd.strip()
'GoDarda'

# Remove trailing whitespace
>>> gd.rstrip()
'    GoDarda'

# Remove leading whitespace
>>> gd.lstrip()
'GoDarda           '
String find() and count() methods
# Find substring index
>>> gd="GoDarda"
>>> gd.find("D",0,len(gd))
2

>>> gd.find("ar",0,len(gd))
3

# Count occurrences of substring
>>> gd.count("a",1,len(gd))
2

>>> gd.count("d",1,len(gd))
1
Strings sort() method
# Sort list in place
>>> gd=["G","o","D","a","r","d","a"]
>>> gd.sort()
>>> gd
['D', 'G', 'a', 'a', 'd', 'o', 'r']

# Tuples are immutable and cannot be sorted in place
>>> gd=("G","o","D","a","r","d","a")
>>> gd.sort()
Traceback (most recent call last):
  File "<stdin>", line 1, in <module>
AttributeError: 'tuple' object has no attribute 'sort'
String startswith() and endswith() methods
# Check if string starts with pattern
>>> gd="GoDarda"
>>> gd.startswith("g")
False

>>> gd.startswith("G")
True

>>> gd.startswith("GoDarda")
True

# Check if string ends with pattern
>>> gd.endswith("GoDarda")
True

>>> gd.endswith("da")
True

>>> gd.endswith("A")
False
String zfill() and replace() method
# Pad string with zeros
>>> gd="GoDarda"
>>> len(gd)
7

>>> gd.zfill(7)
'GoDarda'

>>> gd.zfill(10)
'000GoDarda'

>>> gd.zfill(0)
'GoDarda'

# Replace substring
>>> gd.replace("a","###")
'GoD###rd###'

# Replace first occurrence
>>> gd.replace("D","###",1)
'Go###arda'

>>> gd.replace("a","###",1)
'GoD###rda'
String split() and splitlines() methods
# Split string at line breaks
>>> gd="Go\nDarda"
>>> gd.splitlines()
['Go', 'Darda']

# Split string by separator
>>> gd.split("\n")
['Go', 'Darda']
String functions and operations
# Split string by whitespace
>>> 'Hello, World!'.split()
['Hello,', 'World!']

# Sort characters in string
>>> s1=("Hello, World!")
>>> ''.join(sorted(s1))
' !,HWdellloor'

# Sort and strip whitespace
>>> ''.join(sorted(s1)).strip()
'!,HWdellloor'

# Print sorted characters list
>>> print(sorted(s1))
[' ', '!', ',', 'H', 'W', 'd', 'e', 'l', 'l', 'l', 'o', 'o', 'r']

# Sort unique characters
>>> ''.join(sorted(set(s1)))
' !,HWdelor'

# Remove duplicates preserving order
>>> strings="Kiwi Apple Orange Banana Kiwi Apple Kiwi"
>>> words=strings.split()
>>> print(' '.join(sorted(set(words),key=words.index)))
Kiwi Apple Orange Banana
