# Assign string variables
>>> s1="Go"
>>> s2="Darda"
# Concatenate strings
>>> s3=s1+s2
>>> s3
'GoDarda'

# Check for equality
>>> s1==s2
False

# Check for inequality
>>> s1!=s2
True

# Check if s1 is greater than s2
>>> s1>s2
True

# Check if s1 is less than s2
>>> s1<s2
False

# Check if greater than or equal
>>> s3>=s3
True

# Check if less than or equal
>>> s3<=s3
True

# Check for equality
>>> s3==s3
True

# Check for inequality
>>> s3!=s3
False

# Check if objects are identical
>>> s3 is s3
True

# Check if objects are not identical
>>> s3 is not s3
False
Strings are immutable (one can't alter the contents of the object)
