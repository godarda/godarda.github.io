Number conversion using int(), bin(), oct(), and hex() functions
>>> n=255
# Convert to binary string
>>> print("Binary",bin(n))
Binary 0b11111111

# Convert to octal string
>>> print("Octal",oct(n))
Octal 0o377

# Convert to hexadecimal string
>>> print("Hexadecimal",hex(n))
Hexadecimal 0xff


# Convert binary string to decimal
>>> n="11111111"
>>> print("Decimal",int(n,2))
Decimal 255

# Convert binary string to octal
>>> print("Octal",oct(int(n,2)))
Octal 0o377

# Convert binary string to hexadecimal
>>> print("Hexadecimal",hex(int(n,2)))
Hexadecimal 0xff


# Convert octal string to decimal
>>> n=0o377
>>> print("Decimal",int(n))
Decimal 255

# Convert octal string to hexadecimal
>>> print("Hexadecimal",hex(n))
Hexadecimal 0xff


# Convert hexadecimal string to decimal
>>> n=0Xff
>>> print("Decimal",int(n))
Decimal 255

# Convert hexadecimal string to octal
>>> print("Octal",oct(n))
Octal 0o377

# Convert hexadecimal string to binary
>>> print("Binary",bin(n))
Binary 0b11111111
