mkdir command with options
# List files in current directory
godarda@gd:~$ ls
a.out  gd.cpp   gd.py    Documents  Music     Public     Videos
gd.c   GD.java  Desktop  Downloads  Pictures  Templates  Weekdays

# Print working directory
godarda@gd:~$ pwd
/home/godarda

# Create a new directory
godarda@gd:~$ mkdir Linux
# Change directory
godarda@gd:~$ cd Linux/

godarda@gd:~/Linux$ pwd
/home/godarda/Linux


godarda@gd:~/Linux$ ls
# Create nested directories (parent directories)
godarda@gd:~/Linux$ mkdir -p Flavours/{Ubuntu/{Ubuntu_18.04,Ubuntu_20.04},Fedora/{Fedora_31,Fedora_32}}
godarda@gd:~/Linux$ ls
Flavours

godarda@gd:~/Linux$ cd Flavours/Ubuntu/
godarda@gd:~/Linux/Flavours/Ubuntu$ ls
Ubuntu_18.04  Ubuntu_20.04

# Go up one directory level
godarda@gd:~/Linux/Flavours/Ubuntu$ cd ..
godarda@gd:~/Linux/Flavours$ ls
Fedora  Ubuntu

# Create directory with specific permissions
godarda@gd:~/Linux/Flavours$ mkdir -m a=r KaliLinux
godarda@gd:~/Linux/Flavours$ ls
Fedora  KaliLinux  Ubuntu

godarda@gd:~/Linux/Flavours$ ls -l
total 12
drwxrwxr-x 4 godarda godarda 4096 Jun 13 12:03 Fedora
dr--r--r-- 2 godarda godarda 4096 Jun 13 12:04 KaliLinux
drwxrwxr-x 4 godarda godarda 4096 Jun 13 12:03 Ubuntu

# Create directories with verbose output
godarda@gd:~/Linux/Flavours$ mkdir -v CentOS LinuxMint
mkdir: created directory 'CentOS'
mkdir: created directory 'LinuxMint'

godarda@gd:~/Linux/Flavours$ ls
CentOS  Fedora  KaliLinux  LinuxMint  Ubuntu
rmdir command with options
# Remove an empty directory
godarda@gd:~/Linux/Flavours$ rmdir CentOS/
godarda@gd:~/Linux/Flavours$ ls
Fedora  KaliLinux  LinuxMint  Ubuntu

# Remove a directory recursively
godarda@gd:~/Linux/Flavours$ rm -r KaliLinux/
rm: remove write-protected directory 'KaliLinux/'? y

godarda@gd:~/Linux/Flavours$ ls
Fedora LinuxMint Ubuntu

# Return to home directory
godarda@gd:~/Linux/Flavours$ cd
godarda@gd:~$ ls
a.out  gd.cpp   gd.py    Documents  Linux  Pictures  Templates  Weekdays
gd.c   GD.java  Desktop  Downloads  Music  Public    Videos

# Force remove a directory recursively
godarda@gd:~$ rm -rf Linux/
godarda@gd:~$ ls
a.out  gd.cpp   gd.py    Documents  Music     Public     Videos
gd.c   GD.java  Desktop  Downloads  Pictures  Templates  Weekdays
godarda@gd:~$
