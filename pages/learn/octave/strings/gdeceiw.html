# Create a string variable
octave> gd = "GoDarda"
gd = GoDarda

# Check if string starts with pattern
octave> startsWith(gd, "GO")
ans = 0

octave> startsWith(gd, "Go")
ans = 1

# Check if string ends with pattern
octave> endsWith(gd, "ar")
ans = 0

octave> endsWith(gd, "da")
ans = 1

# Find index of first occurrence
octave> index(gd, "a")
ans = 4

# Find index of last occurrence
octave> rindex(gd, "a")
ans = 7

# Get character indices
octave> unicode_idx(gd)
ans =

   1   2   3   4   5   6   7

# Find substring index
octave> strfind(gd, "DARDA")
ans = [](0x0)

octave> strfind(gd, "Darda")
ans = 3
