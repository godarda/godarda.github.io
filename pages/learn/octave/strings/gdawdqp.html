# Create a string variable
octave> gd = "GoDarda"
gd = GoDarda

# Compare two strings for equality
octave> strcmp(gd, "GoDarda")
ans = 1

# Compare strings case-sensitively
octave> strcmp(gd, "GODARDA")
ans = 0

# Compare the first n characters of strings
octave> strncmp(gd, "Go", 2)
ans = 1

# Compare strings ignoring case
octave> strcmpi("GD#12","gd#12")
ans = 1

octave> strcmpi("GD#12","gd#21")
ans = 0

# Compare the first n characters ignoring case
octave> strncmpi("GD#12","gd",2)
ans = 1

# Replace occurrences of a substring
octave> strrep(gd, "a", "#")
ans = GoD#rd#

# Check if the variable is a string
octave> ischar(gd)
ans = 1

# Compare strings character by character
octave> a="Hello"
a = Hello

octave> b="hello"
b = hello

octave> a==b
ans =

  0  1  1  1  1

octave> b="Hello"
b = Hello

octave> a==b
ans =

  1  1  1  1  1
