# Create a string variable
octave> gd = "Go"
gd = Go

# Concatenate strings
octave> gd = [gd, "Darda"]
gd = GoDarda

# Check variable type
octave> typeinfo(gd)
ans = string

# Access character at specific index
octave> gd(3)
ans = D

# Extract a substring
octave> gd(2:6)
ans = oDard

# Get string length
octave> length(gd)
ans = 7

# Reverse the string
octave> fliplr(gd)
ans = adraDoG

# Convert to uppercase
octave> toupper(gd)
ans = GODARDA

# Convert to lowercase
octave> tolower(gd)
ans = godarda

# Sort characters in the string
octave> sort(gd)
ans = DGaador
