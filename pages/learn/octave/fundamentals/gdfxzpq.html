# Addition
octave> 10+20
ans = 30

# Subtraction
octave> 10-20
ans = -10

# Multiplication
octave> 10*20
ans = 200

# Division
octave> 10/20
ans = 0.5000

# Exponentiation
octave> 2^10
ans = 1024

# Modulo operation
octave> mod(-7,4)
ans = 1

octave> mod(7,4)
ans = 3

# Remainder after division
octave> rem(-7,4)
ans = -3
Arithmetic operations on complex numbers
# Add complex numbers directly
octave> complex(4+5j)+complex(2+6j)
ans =   6 + 11i

# Define complex variables
octave> c1=4+5j
c1 =  4 + 5i

octave> c2=2+6j
c2 =  2 + 6i

# Add complex variables
octave> c1+c2
ans =   6 + 11i

# Subtract complex variables
octave> c1-c2
ans =  2 - 1i

# Multiply complex variables
octave> c1*c2
ans = -22 + 34i

# Divide complex variables
octave> c1/c2
ans =  0.9500 - 0.3500i
round(), floor(), ceil() and fix() functions
# Round to nearest integer
octave> round(2.3)
ans = 2

octave> round(2.8)
ans = 3

octave> round(-2.3)
ans = -2

octave> round(-2.8)
ans = -3


# Round down to nearest integer
octave> floor(2.3)
ans = 2

octave> floor(2.8)
ans = 2

octave> floor(-2.3)
ans = -3

octave> floor(-2.8)
ans = -3


# Round up to nearest integer
octave> ceil(2.3)
ans = 3

octave> ceil(2.8)
ans = 3

octave> ceil(-2.3)
ans = -2

octave> ceil(-2.8)
ans = -2


# Round towards zero
octave> fix(2.3)
ans = 2

octave> fix(2.8)
ans = 2

octave> fix(-2.3)
ans = -2

octave> fix(-2.8)
ans = -2
