# Generate a random number between 0 and 1
octave> rand(1)
ans = 0.7154

# Generate a random number between 0 and 10
octave> rand(1)*10
ans = 8.7886

# Generate a random number between 0 and 100
octave> rand(1)*100
ans = 78.646

# Generate a random number between 0 and 1000
octave> rand(1)*1000
ans = 387.05

# Generate a random integer by rounding
octave> round(rand(1)*1000)
ans = 779
Generate random integer numbers
# Generate a random integer between 1 and 100
octave> randi(100)
ans = 21

# Generate a random integer between 1 and 1000
octave> randi(1000)
ans = 848

# Generate a random integer in a specific range
octave> randi([10,15])
ans = 14

octave> randi([10,15])
ans = 10

# Generate a matrix of random integers
octave> randi(5,5)
ans =

    1   1   4   5   1
    2   5   4   4   3
    2   4   3   2   3
    2   1   2   3   1
    3   5   5   3   2
