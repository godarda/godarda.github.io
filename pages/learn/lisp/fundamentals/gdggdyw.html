String operations
# Print a string literal
[1]> "Hello World"
"Hello World"

# Get the length of the string
[2]> (length "Hello World")
11

# Get the character at index 0
[3]> (char "Hello World"0)
#\H

# Get the character at index 10
[4]> (char "Hello World"10)
#\d

# Return the symbol HELLO
[5]> 'hello
HELLO

# Return a symbol with mixed case using escape
[6]> 'h\ello
|HeLLO|

# Return a symbol preserving case using bars
[7]> 'h|ello|
|Hello|
type-of function
# Check the type of the number 0
[1]> (type-of 0)
BIT

# Check the type of the number 1
[2]> (type-of 1)
BIT

# Check the type of the number 2
[3]> (type-of 2)
(INTEGER 0 281474976710655)

# Check the type of a float literal
[4]> (type-of '1e2)
SINGLE-FLOAT

# Check the type of the constant pi
[5]> (type-of pi)
LONG-FLOAT

# Check the type of the symbol pi
[6]> (type-of 'pi)
SYMBOL

# Check the type of the + variable
[7]> (type-of +)
CONS

# Check the type of the - variable
[8]> (type-of -)
CONS

# Check the type of the * variable
[9]> (type-of *)
SYMBOL

# Check the type of the / variable
[10]> (type-of /)
CONS

# Check the type of a string
[11]> (type-of "Hello World")
(SIMPLE-BASE-STRING 11)

# Check the type of the *package* variable
[12]> (type-of *package*)
PACKAGE
symbol-name function
# Get the name of the symbol 'hello
[1]> (symbol-name 'hello)
"HELLO"

# Get the name of the symbol '|hello|
[2]> (symbol-name '|hello|)
"hello"

# Check if the symbol name equals "Hello"
[3]> (string=(symbol-name'h|ello|)"Hello")
T
eql function
# Check if symbol 'A is eql to 'a
[1]> (eql'A'a)
T

# Check if integer 50 is eql to float 50.0
[2]> (eql 50 50.0)
NIL

# Check if two integers are equal
[3]> (eql 50 50)
T
random function
# Generate a random integer up to 500
[1]> (random 500)
153

# Generate a random float up to 500.0
[2]> (random 500.0)
497.5427
remove function
# Remove all occurrences of 1 from the list
[1]> (remove 1'(4 1 5 1 7 2 9 1 5 2 1))
(4 5 7 2 9 5 2)

# Remove all occurrences of 'r from the list
[2]> (remove 'r'(r a d a r))
(A D A)
