#include 
int main()
{
    int gd = 0, gm = 9;
    initgraph(&gd, &gm, NULL);
    setlinestyle(DASHED_LINE, 5, NORM_WIDTH);
    setcolor(15);
    line(50, 50, 150, 150);
    delay(2000);
    return 0;
}
