interface Operations
{
    interface add
    {
        void add(int x,int y);
    }
    interface sub
    {
        void sub(int x,int y);
    }
}

class GD implements Operations
{
    void add(int x,int y) 
    {
        System.out.println("Addition "+(x+y));
    }
    void sub(int x,int y) 
    {
        System.out.println("Subtraction "+(x-y));
    }
    public static void main(String[] args) 
    {
        GD g=new GD();
        g.add(10,20);
        g.sub(20,10);
    }
}
