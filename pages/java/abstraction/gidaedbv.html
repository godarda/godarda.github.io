Example 1: {{ page.title }}
{{ site.javafile }}
{%- highlight java -%}
interface WhiteBank
{
    default void bonus()
    {
        System.out.println("WhiteBank Festival Bonus: 500");
    }
}
interface GreenBank
{
    default void gift()
    {
        System.out.println("GreenBank Festival Gift: 1000");
    }
}
class GD implements WhiteBank, GreenBank
{
    public static void main(String args[])
    {
        GD g=new GD();
        g.bonus();
        g.gift();
    }
}
{%- endhighlight -%}

Example 2: {{ page.title }}
{{ site.javafile }}
{%- highlight java -%}
interface WhiteBank
{
    default void bonus()
    {
        System.out.println("WhiteBank Festival Bonus: 500");
    }
}
interface GreenBank
{
    default void bonus()
    {
        System.out.println("GreenBank Festival Bonus: 1000");
    }
}
class GD implements WhiteBank, GreenBank
{
    public void bonus()
    {
        WhiteBank.super.bonus();
        GreenBank.super.bonus();
    }
    public static void main(String args[])
    {
        GD g=new GD();
        g.bonus();
    }
}
{%- endhighlight -%}
