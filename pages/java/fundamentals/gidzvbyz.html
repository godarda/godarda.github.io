import java.util.Scanner;
class GD
{
    public static void main(String args[])
    {
        float a,b;
        Scanner sc=new Scanner(System.in);
        System.out.println("———————————————————————————————————————————");
        System.out.println("Program to swap the given numbers");
        System.out.println("———————————————————————————————————————————");
        System.out.print("a = ");
        a=sc.nextFloat();
        System.out.print("b = ");
        b=sc.nextFloat();
        System.out.println("——————————————————————————");
        System.out.println("After swapping...");
        System.out.println("——————————————————————————");
        a+=b; 
        b=a-b;
        a-=b;
        System.out.println("a = "+a+"\nb = "+b);
        System.out.println("———————————————————————————————————————————");
    }       
}

{%- highlight java -%}
import java.util.Scanner;
class GD
{
    public static void main(String args[])
    {
        float a,b,c=0;
        Scanner sc=new Scanner(System.in);
        System.out.println("———————————————————————————————————————————");
        System.out.println("Program to swap the given numbers");
        System.out.println("———————————————————————————————————————————");
        System.out.print("a = ");
        a=sc.nextFloat();
        System.out.print("b = ");
        b=sc.nextFloat();
        System.out.println("——————————————————————————");
        System.out.println("After swapping...");
        System.out.println("——————————————————————————");
        c=a; 
        a=b;
        b=c;;
        System.out.println("a = "+a+"\nb = "+b);
        System.out.println("———————————————————————————————————————————");
    }       
}
