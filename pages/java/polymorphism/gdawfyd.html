{{ page.title }}
{{ site.javafile }}
{%- highlight java -%}
class GD
{
    public static void main()
    {
        System.out.println("Welcome to GoDarda");
    }
    public static void main(String args[])
    {
        System.out.println("Hello, World!");
        main();
    }
}
{%- endhighlight -%}
{{ site.output }}
{{ site.runjava }}
Hello, World!
Welcome to GoDarda
godarda@gd:~$

class GD
{
    public static void main(String args) 
    {
        System.out.println("Welcome to "+args);
    }
    public static void main(String[] args) 
    {
        System.out.println("Hello, World!");
        main("GoDarda");
    }
}
