class KW 
{

    {

        ts.add(36985.336F);

        ts.add(987442.933F);
        ts.add(45.73F);
        ts.add(47.99F);
        ts.add(1.20220F);
        ts.add(12255.00F);
        System.out.println(ts);
        
        System.out.println(ts.ceiling(2500F)); 
        //return least element from the set which is greater than or equal to the given element

    }
