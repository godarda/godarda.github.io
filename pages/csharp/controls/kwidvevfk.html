int n;
Console.Write("Enter a number ");
n = Convert.ToInt32(Console.ReadLine());
if (n % 2 == 0)
{
    Console.WriteLine(n + " is an EVEN number");
}
else
{
    Console.WriteLine(n + " is an ODD number");
}
