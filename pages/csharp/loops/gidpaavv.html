C# program to print the right half star pyramid pattern
{{ site.csfile }}
{%- highlight csharp -%}
int i, j;
for (i = 1; i <= 10; i++)
{
    for (j = 1; j <= i; j++)
    {
        Console.Write("*");
    }
    Console.Write("\n");
}
{%- endhighlight -%}

C# program to print the left half star pyramid pattern
{{ site.csfile }}
{%- highlight csharp -%}
int i, j;
for (i = 1; i <= 10; i++)
{
    for (j = 1; j <= 10 - i; j++)
    {
        Console.Write(" ");
    }
    for (j = 1; j <= i; j++)
    {
        Console.Write("*");
    }
    Console.Write("\n");
}
{%- endhighlight -%}
