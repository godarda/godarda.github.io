{% highlight vb %}
Function Fibonacci(n)
    If (n < 2) Then
        Fibonacci = n
    Else
        Fibonacci = Fibonacci(n-1) + Fibonacci(n-2)
    End If
End Function

For i = 0 To 10
    F = Fibonacci(i)
    print F
Next
{% endhighlight %}
