{%- highlight vb -%}
Set Stack = CreateObject("System.Collections.Stack") 
Stack.Push "Apple"
Stack.Push "Banana"
Stack.Push "Cherry"
Stack.Push "Durian"

For Each item In Stack
    s = s & item & " "
Next
print s
print Stack.Contains("Apple") & vbCrLf
print "Count " & Stack.Count
print "Pop " & Stack.Pop & vbCrLf
print "Peek " & Stack.Peek & vbCrLf 
print Join(Stack.ToArray, ", ") & vbCrLf
{%- endhighlight -%}
