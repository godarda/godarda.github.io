#include 
int main()
{
    int gd = 0, gm = 9, i;
    initgraph(&gd, &gm, NULL);
    for (i = 50; i <= 100; i = i + 15)
    {
        setcolor(15);
        circle(150, 150, i);
        delay(200);
    }
    delay(5000);
    return 0;
}
