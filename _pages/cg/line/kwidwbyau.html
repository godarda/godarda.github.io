#include 
int main()
{
    int gd = 0, gm = 9;
    initgraph(&gd, &gm, NULL);
    setlinestyle(DOTTED_LINE, 5, THICK_WIDTH);
    setcolor(15);
    line(50, 50, 150, 150);
    delay(2000);
    return 0;
}
