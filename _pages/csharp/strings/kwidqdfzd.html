C# program to convert lowercase to an uppercase string using ASCII values

{% highlight csharp %}
string s1 = "Kodingwindow";
string s2 = "";
for (int i = 0; i < s1.Length; i++)
{
    if (s1[i] >= 97 && s1[i] <= 122)
        s2 += (char)(s1[i] - 32);
    else
        s2 += s1[i];
}
Console.WriteLine(s2);
