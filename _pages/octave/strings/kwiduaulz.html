kodingwindow@kw:~$ octave

octave> kw = "Koding"
kw = Koding

octave> kw = [kw, "window"]
kw = Kodingwindow

octave> typeinfo(kw)
ans = string

octave> kw(6)
ans = g

octave> kw(2:11)
ans = odingwindo

octave> length(kw)
ans = 12

octave> fliplr(kw)
ans = wodniwgnidoK

octave> toupper(kw)
ans = KODINGWINDOW

octave> tolower(kw)
ans = kodingwindow

octave> sort(kw)
ans = Kddgiinnooww
