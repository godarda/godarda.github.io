#include 
int main()
{
    int sum = 0;
    int arr[5] = {100, 15, 8, 45, 93};
    for (int i = 0; i < 5; i++)
    {
        sum = sum + arr[i];
    }
    printf("The addition of a given array is %d ", sum);
    return 0;
}
