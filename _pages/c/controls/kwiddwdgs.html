#include 
int main()
{
    float a, b;
    printf("———————————————————————————————————————————");
    printf("\nProgram to check the equality of given numbers");
    printf("\n———————————————————————————————————————————");
    printf("\nEnter the 1st number ");
    scanf("%f", &a);
    printf("\nEnter the 2nd number ");
    scanf("%f", &b);
    if (a < b)
    {
        printf("\n%f is less than %f", a, b);
    }
    else if (a == b)
    {
        printf("\n%f is equal to %f", a, b);
    }
    else
    {
        printf("\n%f is greater than %f", a, b);
    }
    printf("\n———————————————————————————————————————————\n");
    return 0;
}
