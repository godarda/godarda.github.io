import pandas as pd
import matplotlib.pyplot as plot
df=pd.read_csv("/home/kodingwindow/kw.csv")

x=df['account_no']
y=df['amount']

plot.bar(x,y, label="Bank Holders")
plot.xlabel("Account Numbers")
plot.ylabel("Current Amount")

plot.title("New Bank Holders")
plot.show()
