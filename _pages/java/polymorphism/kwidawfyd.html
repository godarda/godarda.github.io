{{ page.title }}
{{ site.javafile }}
{% highlight java %}
class KW
{
    public static void main()
    {
        System.out.println("Welcome to KodingWindow");
    }
    public static void main(String args[])
    {
        System.out.println("Hello, World!");
        main();
    }
}
{% endhighlight %}
{{ site.output }}
{{ site.runjava }}
Hello, World!
Welcome to KodingWindow
kodingwindow@kw:~$

class KW
{
    public static void main(String args) 
    {
        System.out.println("Welcome to "+args);
    }
    public static void main(String[] args) 
    {
        System.out.println("Hello, World!");
        main("KodingWindow");
    }
}
{% endhighlight %}
