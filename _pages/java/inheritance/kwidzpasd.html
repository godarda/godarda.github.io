class Bank
{
    void Bonus()
    {
       System.out.print("Festival Bonus: ");
    }  
}
class WhiteBank extends Bank
{
    void WhiteBankBonus()
    {
        System.out.println("WhiteBank 1500");
    }
}

class KW 
{
    public static void main(String[] args) 
    {
        WhiteBank wb=new WhiteBank();
        wb.Bonus();
        wb.WhiteBankBonus();
    }
}
