class KW
{
    public static void main(String args[])
    {
        int size=Integer.MAX_VALUE;
        try
        {
            String[] s=new String[size];
        }
        catch(OutOfMemoryError e)
        {
            System.out.println("Array overflow");
            System.out.println(e.toString());
        }
    }
}
