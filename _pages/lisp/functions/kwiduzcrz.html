(write-line "———————————————————————————————————————————")
(princ "Enter the radius ")
(setq r(read))
(defconstant PI 3.1416)
(defun Area()
(setq A(* PI r r))
(terpri)(princ "The area of a circle is ")
(write A)
(setq C(* 6.2831 r))
(terpri)(princ "The circumference of a circle is ")
(write C))
(Area)
(terpri)(format t "———————————————————————————————————————————")
