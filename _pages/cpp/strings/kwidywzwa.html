#define SIZE 50
using namespace std;

int i, top = -1;
char stack[SIZE];

bool push(char element)
{
    if (top == (SIZE - 1))
    {
        cout<<"\nStack overflow";
        return false;
    }
    else
    {
        stack[++top] = element;
        return true;
    }
}

char pop()
{
    if (top == -1)
    {
        cout<<"\nStack underflow";
        return 0;
    }
    else
    {
        return stack[top--];
    }
}
