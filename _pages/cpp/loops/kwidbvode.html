C++ program to print the star right half pyramid pattern
{{ site.cppfile }}
{%- highlight cpp -%}
#include 
using namespace std;
int main()
{
    int i, j;
    for (i = 1; i <= 10; i++)
    {
        for (j = 1; j <= i; j++)
        {
            cout<<"*";
        }
        cout<<"\n";
    }
    return 0;
}

C++ program to print the star diamond pattern
{{ site.cppfile }}
{%- highlight cpp -%}
#include 
using namespace std;
int main()
{
    int i, j;
    for (i = 1; i <= 7; i++)
    {
        for (j = 1; j <= 7 - i; j++)
            cout<<" ";
        for (j = 1; j <= i; j++)
        {
            cout<<"*"<<" ";
        }
        cout<<"\n";
    }
    for (i = 6; i >= 1; i--)
    {
        for (j = 1; j <= 6 - i; j++)
            cout<<" ";
        for (j = 1; j <= i; j++)
        {
            cout<<" "<<"*";
        }
        cout<<"\n";
    }
    return 0;
}
