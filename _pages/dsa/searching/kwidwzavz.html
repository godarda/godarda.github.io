def BinarySearch(array,num,first,last):
    if(first>last):
        print("\nElement not found!")
        print("———————————————————————————————————————————")
    else:
        mid=(first+last)//2;
        if(array[mid]==num):
            print("\nElement found at index",mid+1)
            print("———————————————————————————————————————————")
        elif(array[mid]>num):
            BinarySearch(array,num,first,mid-1);
        else:
            BinarySearch(array,num,mid+1,last);
