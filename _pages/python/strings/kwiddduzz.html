Strings Concatenation
>>> s1="Koding"
>>> s2="window"
>>> s1+s1
'KodingKoding'

>>> s1+s2
'Kodingwindow'

>>> s1*5
'KodingKodingKodingKodingKoding'

>>> s1,s2
('Koding', 'window')
Escape Characters
>>> kw="Koding\nwindow"
>>> kw
'Koding\nwindow'

>>> kw="Koding\nwindow"
>>> print(kw)
Koding
window
 
>>> kw=r"Koding\nwindow"
>>> print(kw)
Koding\nwindow
 
>>> print('\"Kodingwindow\"')
"Kodingwindow"

>>> kw="Koding\twindow"
>>> print(kw)
Koding	window
